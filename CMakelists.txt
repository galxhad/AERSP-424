# Specify the minimum CMake version required
cmake_minimum_required(VERSION 3.22)

# Define the project name
project(Homework02)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Specify the relative path to Matplot++ (adjust as needed)
set(MATPLOTPP_PATH "${CMAKE_SOURCE_DIR}/third_party/matplotplusplus")

# Add Matplot++ as a subdirectory (assuming it contains a CMakeLists.txt)
add_subdirectory(${MATPLOTPP_PATH} ${CMAKE_BINARY_DIR}/matplot)

# Include other directories (your project headers)
include_directories(${CMAKE_SOURCE_DIR}/include)

# Collect the source files for each question
set(SOURCES_QUESTION1
    src/AerospaceControlSystem.cpp
    src/AltitudeSensor.cpp
    src/PressureSensor.cpp
    src/TemperatureSensor.cpp
    src/SensorFactory.cpp
    src/Question1main.cpp
)

set(SOURCES_QUESTION2
    src/Question2main.cpp
)

set(SOURCES_QUESTION3
    src/airport_simulator.cpp
    src/Question3main.cpp
)

set(SOURCES_QUESTION4_PART1
    src/Question4part1.cpp
)

set(SOURCES_QUESTION4_PART2
    src/Question4part2.cpp
)

# Define executables for each question
add_executable(Question1 ${SOURCES_QUESTION1})
add_executable(Question2 ${SOURCES_QUESTION2})
add_executable(Question3 ${SOURCES_QUESTION3})
add_executable(Question4_part1 ${SOURCES_QUESTION4_PART1})
add_executable(Question4_part2 ${SOURCES_QUESTION4_PART2})

# Link Matplot++ to the executables
target_link_libraries(Question4_part1 PRIVATE matplot)
target_link_libraries(Question4_part2 PRIVATE matplot)
